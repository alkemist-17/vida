module		:=	statement*
statement	:= 	global
			| 	local
			|	block
global		:=	ident '=' expr
local		:=	'loc' ident '=' expr
block		:=	statement*
ident		:=	['_' | letter]+ ['_' | letter | number]*
expr		:=	prefix
			|	binary
prefix		:=	'not' expr
			|	primary
binary		:=	expr binop expr
primary		:=	'true'
			|	'false'
			|	'nil'
			|	ident
			|	'(' expr ')'
binop		:=	'and'
			|	'or'
